CREATE TABLE Students (
  id VARCHAR(255) PRIMARY KEY,
  Name VARCHAR(255),
  Contact_Number INT,
  Email VARCHAR(255),
  Dept VARCHAR(255),
  Year INT
);

CREATE TABLE Gymkhana (
  id VARCHAR(255) PRIMARY KEY,
  Position VARCHAR(255),
  Email VARCHAR(255)
);

CREATE TABLE Faculty_and_Staff (
  id VARCHAR(255) PRIMARY KEY,
  Name VARCHAR(255),
  Contact_Number INT,
  Email VARCHAR(255)
);

CREATE TABLE Student_Supervisor (
  id VARCHAR(255) PRIMARY KEY,
  sid1 VARCHAR(255),
  sid2 VARCHAR(255),
  sid3 VARCHAR(255),
  FOREIGN KEY (sid1) REFERENCES Students(id),
  FOREIGN KEY (sid2) REFERENCES Students(id),
  FOREIGN KEY (sid3) REFERENCES Students(id)
);

CREATE TABLE Room_Info (
  Room_Number INT PRIMARY KEY,
  AC_Non_AC VARCHAR(255),
  Single_Double VARCHAR(255),
  Price_per_day INT
);

CREATE TABLE Applied_Booking (
  Booking_ID VARCHAR(255) PRIMARY KEY,
  Booked_by_user_id VARCHAR(255),
  Room_Number INT,
  Check_In_Date DATE,
  Check_Out_Date DATE,
  Booked_For_Name VARCHAR(255),
  Booked_For_Email VARCHAR(255),
  Total_Billing INT,
  FOREIGN KEY (Room_Number) REFERENCES Room_Info(Room_Number)
);

CREATE TABLE Confirmed_Booking (
  Booking_ID VARCHAR(255) PRIMARY KEY,
  Booked_by_user_id VARCHAR(255),
  Room_Number INT,
  Check_In_Date DATE,
  Check_Out_Date DATE,
  Booked_For_Name VARCHAR(255),
  Booked_For_Email VARCHAR(255),
  Total_Billing INT,
  FOREIGN KEY (Room_Number) REFERENCES Room_Info(Room_Number)
);

CREATE TABLE Booking_History (
  Booking_ID VARCHAR(255) PRIMARY KEY,
  Room_Number INT,
  Check_In_Date DATE,
  Check_Out_Date DATE,
  Booked_For_Name VARCHAR(255),
  Booked_For_Email VARCHAR(255),
  Total_Billing INT,
  FOREIGN KEY (Room_Number) REFERENCES Room_Info(Room_Number)
);
